```typescript
import { test, expect } from '@playwright/test';

test.describe('Sheet (mobile nav) animation', () => {
  test.use({ viewport: { width: 375, height: 812 } }); // iPhone viewport — Sheet only visible on mobile

  test('opens and closes with animation', async ({ page }) => {
    await page.goto('/');

    // The hamburger menu trigger is inside the mobile nav section of Header.tsx
    // Find the Sheet trigger button — it has a Menu icon from lucide-react
    // Use a more resilient selector: the button that triggers the Sheet
    const trigger = page.locator('button').filter({ has: page.locator('svg.lucide-menu') }).first();
    // Fallback: if lucide class not found, try role-based
    const triggerVisible = await trigger.isVisible().catch(() => false);
    const finalTrigger = triggerVisible
      ? trigger
      : page.getByRole('button', { name: /menu/i });

    await finalTrigger.click();

    // SheetContent uses Radix Dialog — data-state transitions from closed -> open
    const sheetContent = page.locator('[role="dialog"]');
    await expect(sheetContent).toBeVisible({ timeout: 5000 });
    await expect(sheetContent).toHaveAttribute('data-state', 'open');

    // Close the sheet — Radix Sheet has a close button (SheetClose with X icon)
    const closeButton = sheetContent.locator('button[data-radix-dialog-close]').first();
    // Fallback: look for close button by role
    const closeVisible = await closeButton.isVisible().catch(() => false);
    const finalClose = closeVisible
      ? closeButton
      : page.getByRole('button', { name: /close/i });

    await finalClose.click();
    await expect(sheetContent).not.toBeVisible({ timeout: 5000 });
  });
});

test.describe('DropdownMenu animation', () => {
  test('opens and closes with animation', async ({ page }) => {
    await page.goto('/');

    // ModeToggle renders a button that triggers the DropdownMenu
    // The button has an accessible name related to theme toggling
    const trigger = page.getByRole('button', { name: /toggle theme/i });
    await trigger.click();

    // DropdownMenuContent has data-radix-dropdown-menu-content attribute
    const content = page.locator('[data-radix-dropdown-menu-content]');
    await expect(content).toBeVisible({ timeout: 5000 });
    await expect(content).toHaveAttribute('data-state', 'open');

    // Close by pressing Escape — standard Radix behavior
    await page.keyboard.press('Escape');
    await expect(content).not.toBeVisible({ timeout: 5000 });
  });
});

test.describe('MobileActionButtons animation', () => {
  test.use({ viewport: { width: 375, height: 812 } }); // Mobile viewport

  test('is visible on mobile with animation classes', async ({ page }) => {
    await page.goto('/');
    // Wait for page to fully load and animation to complete
    await page.waitForTimeout(1000);

    // MobileActionButtons uses static animation classes:
    // animate-in slide-in-from-bottom-4 duration-700 fade-in
    // It is a fixed-position element at the bottom of the viewport
    const bar = page.locator('.animate-in.slide-in-from-bottom-4');
    await expect(bar).toBeVisible({ timeout: 5000 });

    // Verify the element has the expected animation classes
    await expect(bar).toHaveClass(/animate-in/);
    await expect(bar).toHaveClass(/slide-in-from-bottom-4/);
    await expect(bar).toHaveClass(/fade-in/);
  });
});
```

Key implementation details:
- Sheet test uses mobile viewport (375x812) because the hamburger trigger is mobile-only (per Header.tsx)
- Sheet close uses `[data-radix-dialog-close]` attribute selector for reliability
- DropdownMenu trigger uses `getByRole('button', { name: /toggle theme/i })` matching the ModeToggle accessible name
- MobileActionButtons test checks class presence (static animation, no open/close cycle)
- All timeouts set to 5000ms to accommodate animation durations
- reducedMotion: 'no-preference' is set in playwright.config.ts (Plan 01) so animations actually run

Run the tests:
```bash
npx playwright test
```

If any test fails due to selector issues, read the actual component source again and adjust selectors. The Radix `data-state` attribute and `role="dialog"` are the most stable selectors.
